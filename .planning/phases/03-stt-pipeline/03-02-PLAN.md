    2. Audio handling method (for AudioPipeline callback):
       ```python
       async def on_audio_chunk(self, chunk: AudioChunk) -> None:
           """Called by AudioPipeline for each audio chunk."""
           if chunk.is_speech or self._is_accumulating:
               self._audio_buffer.extend(chunk.data)
       ```

    3. VAD event handling:
       ```python
       async def on_vad_event(self, event: VADEvent) -> None:
           """Called when VAD events occur."""
           if event.type == VADEventType.SPEECH_START:
               self._is_accumulating = True
               self._audio_buffer.clear()
               logger.debug("STT: Started accumulating audio")

           elif event.type == VADEventType.SPEECH_END:
               self._is_accumulating = False
               if len(self._audio_buffer) > 0:
                   await self._process_accumulated_audio()
       ```

    4. Transcription method:
       ```python
       async def _process_accumulated_audio(self) -> None:
           """Transcribe accumulated audio buffer."""
           audio_bytes = bytes(self._audio_buffer)
           self._audio_buffer.clear()

           if len(audio_bytes) < 1600:  # Less than 100ms at 16kHz, skip
               logger.debug("STT: Audio too short, skipping")
               return

           logger.info(f"STT: Processing {len(audio_bytes)} bytes of audio")

           # Run transcription in thread pool to not block event loop
           loop = asyncio.get_event_loop()
           result = await loop.run_in_executor(
               None,
               self.transcriber.transcribe,
               audio_bytes
           )

           if result.text.strip():
               logger.info(f"STT: Transcribed: {result.text}")
               for callback in self._transcription_callbacks:
                   try:
                       await callback(result)
                   except Exception as e:
                       logger.error(f"Transcription callback error: {e}")
       ```

    5. Callback registration:
       - add_transcription_callback(callback: TranscriptionCallback)
       - remove_transcription_callback(callback)
       - add_partial_callback(callback) - for streaming partials (Phase 2 use)
       - remove_partial_callback(callback)

    6. Stats property returning buffer size, is_accumulating, total transcriptions count

    3. Start partial loop when accumulation begins:
       ```python
       async def _start_partial_loop(self) -> None:
           """Periodically transcribe partial audio while speaking."""
           while self._is_accumulating and self.enable_partials:
               await asyncio.sleep(self.partial_interval_ms / 1000)

               if not self._is_accumulating:
                   break

               current_buffer = bytes(self._audio_buffer)
               if len(current_buffer) < 3200:  # At least 200ms
                   continue

               # Run partial transcription
               loop = asyncio.get_event_loop()
               try:
                   result = await loop.run_in_executor(
                       None,
                       self.transcriber.transcribe,
                       current_buffer
                   )
                   if result.text.strip():
                       logger.debug(f"STT partial: {result.text}")
                       for callback in self._partial_callbacks:
                           await callback(result)
               except Exception as e:
                   logger.error(f"Partial transcription error: {e}")
       ```

    4. Update on_vad_event to start/stop partial loop:
       - SPEECH_START: Start partial task if enable_partials
       - SPEECH_END: Cancel partial task if running

    5. Ensure partial loop doesn't interfere with final transcription
       - Final transcription happens after partial task cancelled
